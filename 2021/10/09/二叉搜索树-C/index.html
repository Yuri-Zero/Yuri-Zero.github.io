二叉搜索树-C++

2021年10月09日 1,573 字 大概 10 分钟

对标准库容器进行了粗劣模仿，支持了普通迭代器 常量迭代器 反向迭代器 反向常量迭代器 foreach 和二叉树的基本操作API 以及swap 复制构造 移动构造 和 = 运算符重载

头文件：MyBinarySearchTree.H


#ifndef BINARYSEARCHTREEFINAL_TREE_H
#define BINARYSEARCHTREEFINAL_TREE_H

#include <memory>
#include <iostream>
#include <functional>

template<typename T, typename Compare = std::less<T>, typename Alloc=std::allocator<T>>
class binary_search_tree {
    struct node {
        T element;
        node *m_left_child;
        node *m_right_child;
        node *m_parent;
        node *m_cur = this;

        bool has_two_child() const {
            return this->m_right_child != nullptr && this->m_left_child != nullptr;
        }

        node *next() const {
            auto tmp = this;
            if (tmp->m_right_child) {
                return find_min_node(tmp->m_right_child);
            } else {
                node *parent = tmp->m_parent;
                while (parent != nullptr && parent->m_left_child != tmp) {
                    tmp = parent;
                    parent = tmp->m_parent;
                }
                return parent;
            };
        }

        node *prev() const {
            auto tmp = this;
            if (tmp->m_right_child) {
                return find_max_node(tmp->m_left_child);
            } else {
                while (tmp->m_parent && tmp->m_parent->m_right_child) {
                    tmp = tmp->m_parent;
                    if (tmp == tmp->m_parent->m_right_child) {
                        return tmp->m_parent;
                    }
                }
            }
        }

        node *find_min_node(node *cur) const {
            auto tmp = cur;
            while (tmp->m_left_child) {
                tmp = tmp->m_left_child;
            }
            return tmp;
        }

        node *find_max_node(node *cur) const {
            auto tmp = cur;
            while (tmp->m_right_child) {
                tmp = tmp->m_right_child;
            }
            return tmp;
        }

        bool has_child() const {
            return this->m_right_child || this->m_left_child;
        }
    };

    struct header_node {
        node *parent; // root;
        node *left;// min;
        node *right;// max;
    };

    struct iterator {
        binary_search_tree<T, Compare, Alloc> *tree;
        node *cur;

        iterator(node *node, binary_search_tree *tree) {
            cur = node;
            this->tree = tree;
        }

        iterator &operator++() {
            if (cur != tree->header.parent) {
                auto tmp = cur->next();
                cur = cur->next();
            } else {
                cur = tree->header.parent->next();
            }
            return *this;
        }

        iterator &operator--() {
            if (cur != tree->header.parent) {
                cur = cur->prev();
            } else {
                cur = tree->header.parent->prev();
            }
            return *this;
        }

        iterator operator++(int) {
            auto old = *this;
            ++*this;
            return old;
        }

        iterator operator--(int) {
            auto old = *this;
            --*this;
            return old;
        }

        T &operator*() const {
            return cur->element;
        }

        node *operator->() {
            return cur;
        }

        bool operator==(const iterator &rhs) const {
            return this->cur == rhs.cur;
        }

        bool operator!=(const iterator &rhs) const {
            return !operator==(rhs);
        }

        node *iter_find_node(node *node) {
            auto tmp = node;
            while (tmp->m_left_child) {
                tmp = tmp->m_left_child;
            }
            return tmp;
        }
    };

    struct const_iterator {
        const binary_search_tree<T, Compare, Alloc> *tree;
        const node *cur;

        const_iterator(const node *n, const binary_search_tree *tree) {
            cur = n;
            this->tree = tree;
        }

        const_iterator(const_iterator const &) = default;

        const_iterator(iterator const &rhs) {
            this->cur = rhs.cur;
            this->tree = rhs.tree;
        }

        const_iterator &operator++() {
            if (cur != tree->header.parent) {
                auto tmp = cur->next();
                cur = cur->next();
            } else {
                cur = tree->header.parent->next();
            }
            return *this;
        }

        const_iterator &operator--() {
            if (cur != tree->header.parent) {
                cur = cur->prev();
            } else {
                cur = tree->header.parent->prev();
            }
            *this;
        }

        const_iterator operator++(int) {
            auto old = *this;
            ++*this;
            return old;
        }

        const_iterator operator--(int) {
            auto old = *this;
            --*this;
            return old;
        }

        const T &operator*() const {
            return cur->element;
        }

        // node* operator ->() const{
        //     return cur;
        // }
        bool operator==(const_iterator &rhs) {
            return this->cur == rhs.cur;
        }

        bool operator!=(const_iterator &rhs) {
            return !operator==(rhs);
        }

        node *iter_find_node(node *node) {
            auto tmp = node;
            while (tmp->m_left_child) {
                tmp = tmp->m_left_child;
            }
            return tmp;
        }
    };

public:
    using value_type = T;
    using allocator_type = Alloc;
    using alloc_traits = std::allocator_traits<allocator_type>;
    using pointer = typename alloc_traits::pointer;
    using const_pointer = typename alloc_traits::const_pointer;
    using size_type = typename alloc_traits::size_type;
    using difference_type = typename alloc_traits::difference_type;
    using node_allocator = typename std::allocator_traits<Alloc>::template rebind_alloc<node>;
    using propagate_on_container_copy_assignment
    = typename std::allocator_traits<allocator_type>
    ::propagate_on_container_copy_assignment;
    using propagate_on_container_move_assignment = typename std::allocator_traits<allocator_type>
    ::propagate_on_container_move_assignment;
    using propagate_on_container_swap = typename std::allocator_traits<allocator_type>
    ::propagate_on_container_swap;
private:
    header_node header;
    Compare cmp = Compare();
    size_t Size;
    node_allocator m_alloc;
public:
    binary_search_tree() {
        header.parent = header.left = header.right = nullptr;
        Size = 0;
    };

    binary_search_tree(const binary_search_tree &rhs) {
        this->cmp = rhs.cmp;
        this->m_alloc = std::allocator_traits<node_allocator>::select_on_container_copy_construction(rhs.m_alloc);
        copy(this->header.parent, rhs.header.parent);
        header.right = FindMaxNode(header.parent);
        header.left = FindMinNode(header.parent);
    }

    binary_search_tree(binary_search_tree &&rhs) noexcept {
        cmp = std::move(rhs.cmp);
        Size = rhs.Size;
        header = std::move(rhs.header);
        m_alloc = std::move(rhs.m_alloc);
    }

    binary_search_tree &operator=(const binary_search_tree &rhs) {
        if (std::addressof(rhs) != this) {
            if constexpr (propagate_on_container_copy_assignment()) {
                this->m_alloc = rhs.m_alloc;
            }
            copy(this->header.parent, rhs.header.parent);
            header.right = FindMaxNode(header.parent);
            header.left = FindMinNode(header.parent);
        }
    }

    binary_search_tree &operator=(binary_search_tree &&rhs) noexcept {
        if (std::addressof(rhs) != this) {
            if (propagate_on_container_move_assignment()) {
                cmp = std::move(rhs.cmp);
                Size = rhs.Size;
                header = std::move(rhs.header);
                m_alloc = std::move(rhs.m_alloc);
            } else if (typename std::allocator_traits<node_allocator>::is_always_equal() ||
                       this->m_alloc == rhs.m_alloc) {
                cmp = std::move(rhs.cmp);
                Size = rhs.Size;
                header = std::move(rhs.header);
            } else {
                copy(this->header.parent, rhs.header.parent);
            }
        }
    }

    ~binary_search_tree() {
        clear();
        this->header.parent = nullptr;
        this->header.left = nullptr;
        this->header.right = nullptr;
    }

    bool empty() const {
        return header.parent == nullptr;
    }

    size_t size() const {
        return this->Size;
    }
    node *getroot() const {
        return header.parent;
    }
    iterator begin() { return iterator(minhelper(), this); }

    iterator end() { return iterator(maxhelper(), this); }

    const_iterator begin() const { return min(); }

    const_iterator end() const { return max(); }

    const_iterator cbegin() const { return begin(); }

    const_iterator cend() const { return end(); }

    std::reverse_iterator<iterator> rbegin() { return std::reverse_iterator<iterator>(end()); }

    std::reverse_iterator<iterator> rend() { return std::reverse_iterator<iterator>(begin()); }

    std::reverse_iterator<const_iterator> rbegin() const { return std::reverse_iterator<const_iterator>(end()); }

    std::reverse_iterator<const_iterator> rend() const { return std::reverse_iterator<const_iterator>(begin()); }

    std::reverse_iterator<const_iterator> rcbegin() const { return rbegin(); }

    std::reverse_iterator<const_iterator> rcend() const { return rend(); }

    iterator insert(const T &value) { return iterator(inserthelper(value), this); }

    iterator insert(T &&value) { return iterator(inserthelper(value), this); }

    iterator remove(const T &value) { return iterator(removehelper(value), this); }

    iterator find(const T &target) { return iterator(findhelper(target), this); }

    const_iterator max() const { return const_iterator(maxhelper(), this); }

    const_iterator min() const { return const_iterator(minhelper(), this); }

    iterator upper_bound(const T &value) { return iterator(upper_boundhelper(value), this); }

    iterator lower_bound(const T &value) { return iterator(lower_boundhelper(value), this); }

    void clear() {
        if (Size > 0) {
            earse_since(header.parent);
            Size = 0;
        }
    }

    void swap(binary_search_tree &rhs) {
        if (std::addressof(rhs) != this) {
            std::swap(rhs.header, this->header);
            std::swap(rhs.Size, this->Size);
            std::swap(rhs.cmp, this->cmp);
        }
    }

private:
    node *create_new_node(T value, node *parent = nullptr) {
        auto new_node = this->m_alloc.allocate(sizeof(node));
        std::construct_at(std::addressof(new_node->element),value);
        new_node->m_parent = parent;
        new_node->m_left_child = new_node->m_right_child = nullptr;
        return new_node;
    }

    node *FindMinNode(node *ptr) {
        node *tmp = ptr;
        if (tmp == NULL) {
            return NULL;
        }
        while (tmp->m_left_child != NULL) {
            tmp = tmp->m_left_child;
        }
        return tmp;
    }

    node *FindMaxNode(node *ptr) {
        node *tmp = ptr;
        if (tmp == nullptr) {
            return nullptr;
        }
        while (tmp->m_right_child != nullptr) {
            tmp = tmp->m_right_child;
        }
        return tmp;
    }

    node *maxhelper() const {
        return header.right->m_right_child;
    }

    node *minhelper() const {
        return header.left;
    }

    node *inserthelper(const T& value) {
        if (empty()) {
            header.parent = create_new_node(value);
            header.left = header.right = header.parent;
            return header.parent;
        } else {
            auto tmp = header.parent;
            auto parent = header.parent;
            while (tmp) {
                if (!cmp(tmp->element, value)) {
                    parent = tmp;
                    tmp = tmp->m_left_child;
                } else{
                    parent = tmp;
                    tmp = tmp->m_right_child;
                }
            }
            if (cmp(parent->element, value)) {
                parent->m_right_child = create_new_node(value, parent);
                if (header.right->element < parent->m_right_child->element) {
                    header.right = parent->m_right_child;
                }
                Size++;
                return parent->m_right_child;
            } else  {
                if (parent->element != value) {
                    parent->m_left_child = create_new_node(value, parent);
                    if (header.left->element > parent->m_left_child->element) {
                        header.left = parent->m_left_child;
                    }
                    Size++;
                    return parent->m_left_child;
                }
            }
        }
        return nullptr;
    };

    node *inserthelper(T &&value) {
        return inserthelper(value);
    }

    node *removehelper(const T &value) {
        node *target = this->findhelper(value);
        node *parent = target->m_parent;
        node *del = target;
        if (target == nullptr) {
            return nullptr;
        }
        if (target == minhelper()) {
            header.left = target->next();
        }
        if (target == maxhelper()) {
            header.right = target->next();
        }
        //当没有子树时——叶子节点
        if (!target->has_child()) {
//
            if (target == header.parent) {
                header.parent = nullptr;
            } else if (parent->m_left_child == target) {
                parent->m_left_child = nullptr;
            } else if (parent->m_right_child == target) {
                parent->m_right_child = nullptr;
            }
        } else if (target->has_child() && !target->has_two_child()) {
            if (target == header.parent) {
                if (target->m_left_child) {
                    header.parent = target->m_left_child;
                } else {
                    header.parent = target->m_right_child;
                }
            } else {
                if (parent->m_left_child == target) {
                    if (target->m_left_child != nullptr) {
                        parent->m_left_child = target->m_left_child;
                    } else if (target->m_right_child != nullptr) {
                        parent->m_left_child = target->m_right_child;
                    }
                } else {
                    if (target->m_left_child != nullptr) {
                        parent->m_right_child = target->m_left_child;
                    } else {
                        parent->m_right_child = target->m_right_child;
                    }
                }
            }
        } else if (target->has_two_child()) {
            target->element = target->next()->element;
            del = target->next();
            if (target->next() == target->next()->m_parent->m_left_child) {
                node *ttmp = del;
                ttmp->m_parent->m_left_child = nullptr;
            } else if (target->next() == target->next()->m_parent->m_right_child) {
                node *ttmp = del;
                ttmp->m_parent->m_right_child = nullptr;
            }
        }
        std::destroy_at(del);
        this->m_alloc.deallocate(del, sizeof(node));
    }

    node *findhelper(const T &target) {
        auto tmp = header.parent;
        while (tmp) {
            if (tmp->element == target) {
                return tmp;
            } else if (cmp(tmp->element, target)) {
                tmp = tmp->m_right_child;
            } else if (!cmp(tmp->element, target)) {
                tmp = tmp->m_left_child;
            }
        }
        return nullptr;
    }

    node *lower_boundhelper(const T &value) {
        auto tmp = header.parent;
        while (tmp) {
            if (!cmp(tmp->element,value)||tmp->element == value) {
                return tmp;
            } else {
                tmp = tmp->m_right_child;
            }
        }
        return nullptr;
    }

    node *upper_boundhelper(const T &value) {
        auto tmp = header.parent;
        while (tmp) {
            if (cmp(tmp->element,value)||tmp->element == value) {
                return tmp;
            } else {
                tmp = tmp->m_right_child;
            }
        }
        return nullptr;
    }

    void copy(node *&lsh, node *rsh, node *parent = nullptr) {
        if (rsh == nullptr) {
            return;
        }
        lsh = create_new_node(rsh->element, parent);
        copy(lsh->m_left_child, rsh->m_left_child, lsh);
        copy(lsh->m_right_child, rsh->m_right_child, lsh);
    }

    void destory_node(node *p) {
        std::destroy_at(std::addressof(p->element));
        this->m_alloc.deallocate(p, sizeof(node));
    }

    void earse_since(node *&p) {
        if(p == nullptr){
            return;
        }
        earse_since(p->m_left_child);
        earse_since(p->m_right_child);
        destory_node(p);
        p = nullptr;
    }
};

#endif //BINARYSEARCHTREEFINAL_TREE_H
