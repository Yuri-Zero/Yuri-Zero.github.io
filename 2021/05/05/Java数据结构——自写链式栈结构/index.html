Java数据结构——自写链式栈结构

2021年05月05日 591 字 大概 3 分钟

这篇文章主要留作存档，是一篇数据结构——链表的代码文，这篇的数据结构为链式栈。


参考书目:《数据结构与算法分析Java语言描述》
本文包括一个链式栈类,它叫做LinkedStack,其内部又包括一个迭代器类MyLinkedStackIteratorClass【btw,其实这个迭代器内部类是位于Iterator方法内的，所以实际是属于该方法的…】和一个节点类Node
此栈结构为链式栈，支持入栈【push】
和出栈【pop】返回栈顶元素【get】等操作，以及支持迭代器操作和foreach



以下为代码

import java.util.EmptyStackException;
import java.util.Iterator;

/**
 * 链表栈的主类
 * @author Zarathos
 * @date 2021/05/04
 */
public class LinkedStack<Anytype> implements Iterable<Anytype>{
    /**
     * 内部节点类
     * @author Zarathos
     * @date 2021/05/04
     * @param <Anytype>
     */
    private class Node<Anytype> {
        /**
         * 数据域
         */
        Anytype data;
        /**
         * 节点
         */
        Node<Anytype> next;
        Node(Anytype data,Node<Anytype>next){
            this.data = data;
            this.next = next;
        }
    }
    private Node<Anytype> Head;
    private Node<Anytype> Tail;
    private int theSize;
    private int modCount = 0;
    /**
     * 构造器
     */
    LinkedStack(){doclear();}

    /**
     * 清空栈
     */
    public void clear(){doclear();}

    /**
     * 用于清空或构造新栈
     */
    private void doclear(){
        Head = new Node<Anytype>(null,null);
        Tail = new Node<Anytype>(null,null);
        Head.next=Tail;
        theSize=0;
        modCount++;
    }

    /**
     * 入栈
     * @param data 填入元素 类型为Anytype
     */
    public void push(Anytype data){
        var p=new Node<Anytype>(data,null);
        p.next=Head.next;
        Head.next=p;
        theSize++;
        modCount++;
    }

    /**
     * 查找
     * @return Anytype 栈顶元素
     */
    public Anytype get(){
        if(isEmpty()){
            throw new EmptyStackException();
        }
        else{
            return Head.next.data;
        }
    }

    /**
     * 出栈
     * @return 被删除的原栈顶的值
     */
    public Anytype pop(){
        if(isEmpty()){
            throw new EmptyStackException();
        }
        var p=Head.next;
        Anytype oldVal=p.data;
        Head.next=p.next;
        modCount++;
        theSize--;
        return oldVal;
    }
    /**
     * 返回元素个数
     * @return int 当前元素个数
     */
    public int Size(){return theSize;}

    /**
     * 返回是否为空
     * @return boolean true/false
     */
    public boolean isEmpty(){return theSize<0;}

    /**
     * 迭代器
     * @return Iterator<Anytype> 迭代器
     */
    @Override
    public Iterator<Anytype> iterator(){
        /**
         * 内部迭代器类
         * @author Zarathos
         * @date 2021/05/04
         */
        class MyLinkedStackIteratorClass implements Iterator<Anytype>{
            private Node<Anytype>current=Head.next;
            private int expectedModCount=modCount;
            private boolean okToRemove=false;
            @Override
            public boolean hasNext(){return current!=Tail;}
            @Override
            public Anytype next(){
                if(modCount!=expectedModCount){
                    throw new java.util.ConcurrentModificationException();
                }
                if(!hasNext()){
                    throw new java.util.NoSuchElementException();
                }
                Anytype nextItem=current.data;
                current=current.next;
                okToRemove=true;
                return  nextItem;
            }
            @Override
            public void remove(){
                if(modCount!=expectedModCount){
                    throw new java.util.ConcurrentModificationException();
                }
                if(!okToRemove){
                    throw new IllegalStateException();
                }
                LinkedStack.this.pop();
                expectedModCount++;
                okToRemove=false;
            }
        }
        return new MyLinkedStackIteratorClass();
    }
}
